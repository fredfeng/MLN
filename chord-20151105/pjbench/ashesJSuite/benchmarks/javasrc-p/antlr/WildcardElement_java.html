33	class WildcardElement extends AlternativeElement {
34		protected String label;
35	
36	
37		public WildcardElement(Grammar g, Token t, int autoGenType) {
38			super(g, autoGenType);
39			line = t.getLine();
40		}
41		public void generate() {
42			grammar.generator.gen(this);
43		}
44		public String getLabel() {
45			return label;
46		}
47		public Lookahead look(int k) {
48			return grammar.theLLkAnalyzer.look(k, this);
49		}
50		public void setLabel(String label_) { 
51			label = label_; 
52		}
53		public String toString() {
54			String s = " ";
55			if ( label!=null ) s += label+":";
56			return s + ".";
57		}
58	}
