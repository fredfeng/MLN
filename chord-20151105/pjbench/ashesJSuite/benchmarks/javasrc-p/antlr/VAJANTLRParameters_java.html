1	package antlr;
2	
3	import java.io.Serializable;
4	
5	/** A simple class to store some runtime parameters in visualage
6	 */
7	public class VAJANTLRParameters implements Serializable {
8		private String grammarFile;
9		private boolean debug;
10		private boolean writeToFile;
11	
12	
13		public VAJANTLRParameters(String file, boolean d, boolean w) {
14			grammarFile = file;
15			debug = d;
16			writeToFile = w;
17		}
18		public boolean getDebug() {return debug;}
19		public String getGrammarFile() {return grammarFile;}
20		public static String getKey(String packageName) {
21			return "antlr-vaj." + packageName;
22		}
23		public boolean getWriteToFile() {return writeToFile;}
24	}
