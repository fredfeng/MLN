33	/**A GrammarAtom is either a token ref, a character ref, or string.
34	 * The analysis doesn't care.
35	 */
36	abstract class GrammarAtom extends AlternativeElement {
37		protected String label;
38		protected String atomText;
39		protected int tokenType = Token.INVALID_TYPE;
40		protected boolean not = false;	// ~T or ~'c' or ~"foo"
41	
42	
43		public GrammarAtom(Grammar g, Token t, int autoGenType) {
44			super(g, autoGenType);
45			atomText = t.getText();
46		}
47		public String getLabel() {
48			return label;
49		}
50		public void setLabel(String label_) { 
51			label = label_; 
52		}
53		public String toString() {
54			String s = " ";
55			if ( label!=null ) s += label+":";
56			if ( not ) s += "~";
57			return s+atomText;
58		}
59	}
