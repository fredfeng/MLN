33	import antlr.collections.AST;
34	
35	/** Common AST node implementation */
36	public class CommonAST extends BaseAST {
37		int ttype = Token.INVALID_TYPE;
38		String text;
39	
40	
41		/** Get the token text for this node */
42		public String getText() { return text; }
43		/** Get the token type for this node */
44		public int getType() { return ttype; }
45		public void initialize(int t, String txt) {
46			setType(t);
47			setText(txt);
48		}
49	public void initialize(AST t) {
50		setText(t.getText());
51		setType(t.getType());
52	}
53	public void initialize(Token tok) {
54		setText(tok.getText());
55		setType(tok.getType());
56	}
57		/** Set the token text for this node */
58		public void setText(String text_) { 
59			text = text_; 
60		}
61		/** Set the token type for this node */
62		public void setType(int ttype_) { 
63			ttype = ttype_; 
64		}
65	}
